    5. Replace handlePurchase mock with real Stripe redirect:
       const handlePurchase = async (planId: string) => {
         setIsProcessing(true)
         try {
           const response = await fetch('/api/stripe/create-checkout-session', {
             method: 'POST',
             headers: { 'Content-Type': 'application/json' },
             body: JSON.stringify({ planId }),
           })
           const { url, error } = await response.json()
           if (error || !url) {
             setError('Unable to start checkout. Please try again.')
             return
           }
           window.location.href = url
         } catch {
           setError('Network error. Please try again.')
         } finally {
           setIsProcessing(false)
         }
       }

    6. Update CreditPurchaseModalProps:
       - Remove onPurchase prop (no longer needed — Stripe handles the result via webhook)
       - Keep isOpen: boolean and onClose: () => void

    8. Add data-testid="credit-modal-plan-card" to each plan card for future E2E tests.

    Keep Coins icon in title.
